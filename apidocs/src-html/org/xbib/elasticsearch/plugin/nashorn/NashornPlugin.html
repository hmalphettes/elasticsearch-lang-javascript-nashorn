019package org.xbib.elasticsearch.plugin.nashorn;
020
021import org.elasticsearch.plugins.AbstractPlugin;
022import org.elasticsearch.script.ScriptModule;
023import org.xbib.elasticsearch.script.nashorn.NashornScriptEngineService;
024
025/**
026 *
027 */
028public class NashornPlugin extends AbstractPlugin {
029
030    @Override
031    public String name() {
032        return "lang-nashorn";
033    }
034
035    @Override
036    public String description() {
037        return "Nashorn Plugin for Javascript ECMA scripting support";
038    }
039
040    public void onModule(ScriptModule module) {
041        module.addScriptEngine(NashornScriptEngineService.class);
042    }
043}
