019package org.xbib.elasticsearch.script.nashorn;
020
021import jdk.nashorn.internal.objects.NativeArray;
022import jdk.nashorn.internal.runtime.ConsString;
023import jdk.nashorn.internal.runtime.ScriptObject;
024import jdk.nashorn.internal.runtime.Undefined;
025import jdk.nashorn.internal.runtime.arrays.ArrayData;
026
027import java.util.ArrayList;
028import java.util.LinkedHashMap;
029import java.util.Map;
030
031public class NashornUnwrapper {
032
033    private NashornUnwrapper() {
034    }
035
036    /**
037     * Convert an object from a script wrapper value to a serializable value valid outside
038     * of the Nashorn script processor context.
039     *
040     * This includes converting Array objects to Lists of valid objects.
041     *
042     * @param value the value to convert from script wrapper object to external object value
043     * @return unwrapped and converted value
044     */
045    public static Object unwrapValue(Object value) {
046        if (value == null) {
047            return null;
048        }
049        if (value instanceof NativeArray) {
050            NativeArray nativeArray = (NativeArray)value;
051            ArrayData data = nativeArray.getArray();
052            int length = (int)data.length();
053            ArrayList<Object> list = new ArrayList<Object>(length);
054            for (int i = 0; i < length; i++) {
055                list.add(unwrapValue(data.getObject(i)));
056            }
057            return list;
058        } else if (value instanceof Map) {
059            Map map = (Map)value;
060            Map<Object,Object> result = new LinkedHashMap<Object,Object>();
061            for (Object key : map.keySet()) {
062                result.put(key, unwrapValue(map.get(key)));
063            }
064            return result;
065        } else if (value instanceof ScriptObject) {
066            ScriptObject object = (ScriptObject)value;
067            Map<Object,Object> result = new LinkedHashMap<Object,Object>();
068            for (Object key : object.getOwnKeys(true)) {
069                result.put(key, unwrapValue(object.get(key)));
070            }
071            return result;
072        } else if (value instanceof Undefined) {
073            return null;
074        } else if (value instanceof ConsString) {
075            return value.toString();
076        } else {
077            return value;
078        }
079    }
080
081}
