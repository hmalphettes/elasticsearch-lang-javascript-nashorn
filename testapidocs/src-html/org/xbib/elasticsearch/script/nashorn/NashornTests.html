001package org.xbib.elasticsearch.script.nashorn;
002
003import org.junit.Test;
004
005import javax.script.Bindings;
006import javax.script.Compilable;
007import javax.script.CompiledScript;
008import javax.script.ScriptContext;
009import javax.script.ScriptEngine;
010import javax.script.ScriptEngineManager;
011import javax.script.ScriptException;
012import java.util.HashMap;
013import java.util.Map;
014
015import static org.hamcrest.MatcherAssert.assertThat;
016import static org.hamcrest.Matchers.equalTo;
017
018public class NashornTests {
019
020    @Test
021    public void intArrayLengthTest() throws ScriptException {
022        Map<String, Object> vars = new HashMap<String, Object>();
023        Integer[] l = new Integer[] { 1,2,3 };
024        vars.put("l", l);
025        String script = "l.length";
026        ScriptEngineManager manager = new ScriptEngineManager();
027        ScriptEngine engine = manager.getEngineByName("nashorn");
028        ScriptContext context = engine.getContext();
029        Bindings bindings = context.getBindings(ScriptContext.ENGINE_SCOPE);
030        bindings.putAll(vars);
031        Compilable compilable = (Compilable)engine;
032        CompiledScript compiledScript = compilable.compile(script);
033        Object o = compiledScript.eval();
034        assertThat(((Number) o).intValue(), equalTo(3));
035    }
036
037    @Test
038    public void objectArrayTest() throws ScriptException {
039        Map<String, Object> vars = new HashMap<String, Object>();
040        final Map<String, Object> obj2 = new HashMap<String,Object>() {{
041            put("prop2", "value2");
042        }};
043        final Map<String, Object> obj1 = new HashMap<String,Object>() {{
044            put("prop1", "value1");
045            put("obj2", obj2);
046        }};
047        vars.put("l", new Object[] { "1", "2", "3", obj1 });
048        String script = "l.length";
049        ScriptEngineManager manager = new ScriptEngineManager();
050        ScriptEngine engine = manager.getEngineByName("nashorn");
051        Bindings bindings = engine.getBindings(ScriptContext.ENGINE_SCOPE);
052        bindings.putAll(vars);
053        Compilable compilable = (Compilable)engine;
054        CompiledScript compiledScript = compilable.compile(script);
055        Object o = compiledScript.eval(bindings);
056        assertThat(((Number) o).intValue(), equalTo(4));
057    }
058}
