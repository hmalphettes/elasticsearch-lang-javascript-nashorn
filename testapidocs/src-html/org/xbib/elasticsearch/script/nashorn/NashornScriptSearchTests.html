030    @Test
031    public void testNashornFilter() throws Exception {
032        wipeIndices("test");
033        createIndex("test");
034        index("test", "type1", "1", jsonBuilder().startObject().field("test", "value beck").field("num1", 1.0f).endObject());
035        flush();
036        index("test", "type1", "2", jsonBuilder().startObject().field("test", "value beck").field("num1", 2.0f).endObject());
037        flush();
038        index("test", "type1", "3", jsonBuilder().startObject().field("test", "value beck").field("num1", 3.0f).endObject());
039        refresh();
040
041        logger.info(" --> running doc['num1'].value > 1");
042        SearchResponse response = client().prepareSearch()
043                .setQuery(filteredQuery(matchAllQuery(), scriptFilter("doc['num1'].value > 1").lang("nashorn")))
044                .addSort("num1", SortOrder.ASC)
045                .addScriptField("sNum1", "nashorn", "doc['num1'].value", null)
046                .execute().actionGet();
047
048        assertThat(response.getHits().totalHits(), equalTo(2l));
049
050        logger.debug("---> got doc0={}, doc1={}",
051                response.getHits().getAt(0).fields().get("sNum1").values().get(0),
052                response.getHits().getAt(1).fields().get("sNum1").values().get(0) );
053
054        assertThat(response.getHits().getAt(0).id(), equalTo("2"));
055        assertThat((Double) response.getHits().getAt(0).fields().get("sNum1").values().get(0), equalTo(2.0));
056        assertThat(response.getHits().getAt(1).id(), equalTo("3"));
057        assertThat((Double) response.getHits().getAt(1).fields().get("sNum1").values().get(0), equalTo(3.0));
058
059        logger.info(" --> running doc['num1'].value > param1");
060        response = client().prepareSearch()
061                .setQuery(filteredQuery(matchAllQuery(), scriptFilter("doc['num1'].value > param1").lang("nashorn").addParam("param1", 2)))
062                .addSort("num1", SortOrder.ASC)
063                .addScriptField("sNum1", "nashorn", "doc['num1'].value", null)
064                .execute().actionGet();
065
066        assertThat(response.getHits().totalHits(), equalTo(1l));
067        assertThat(response.getHits().getAt(0).id(), equalTo("3"));
068        assertThat((Double) response.getHits().getAt(0).fields().get("sNum1").values().get(0), equalTo(3.0));
069
070        logger.info(" --> running doc['num1'].value > param1");
071        response = client().prepareSearch()
072                .setQuery(filteredQuery(matchAllQuery(), scriptFilter("doc['num1'].value > param1").lang("nashorn").addParam("param1", -1)))
073                .addSort("num1", SortOrder.ASC)
074                .addScriptField("sNum1", "nashorn", "doc['num1'].value", null)
075                .execute().actionGet();
076
077        assertThat(response.getHits().totalHits(), equalTo(3l));
078        assertThat(response.getHits().getAt(0).id(), equalTo("1"));
079        assertThat((Double) response.getHits().getAt(0).fields().get("sNum1").values().get(0), equalTo(1.0));
080        assertThat(response.getHits().getAt(1).id(), equalTo("2"));
081        assertThat((Double) response.getHits().getAt(1).fields().get("sNum1").values().get(0), equalTo(2.0));
082        assertThat(response.getHits().getAt(2).id(), equalTo("3"));
083        assertThat((Double) response.getHits().getAt(2).fields().get("sNum1").values().get(0), equalTo(3.0));
084    }

086    @Test
087    public void testScriptFieldUsingSource() throws Exception {
088        wipeIndices("test");
089        createIndex("test");
090        index("test", "type1", "1",
091                jsonBuilder().startObject()
092                        .startObject("obj1").field("test", "something").endObject()
093                        .startObject("obj2").startArray("arr2").value("arr_value1").value("arr_value2").endArray().endObject()
094                        .endObject());
095        refresh();
096
097        SearchResponse response = client().prepareSearch()
098                .setQuery(matchAllQuery())
099                .addScriptField("s_obj1", "nashorn", "_source.obj1", null)
100                .addScriptField("s_obj1_test", "nashorn", "_source.obj1.test", null)
101                .addScriptField("s_obj2", "nashorn", "_source.obj2", null)
102                .addScriptField("s_obj2_arr2", "nashorn", "_source.obj2.arr2", null)
103                .execute().actionGet();
104
105        Map<String, Object>  sObj1 = (Map<String, Object>) response.getHits().getAt(0).field("s_obj1").value();
106        assertThat(sObj1.get("test").toString(), equalTo("something"));
107        assertThat(response.getHits().getAt(0).field("s_obj1_test").value().toString(), equalTo("something"));
108
109        Map<String, Object> sObj2 = (Map<String, Object>) response.getHits().getAt(0).field("s_obj2").value();
110        List sObj2Arr2 = (List) sObj2.get("arr2");
111        assertThat(sObj2Arr2.size(), equalTo(2));
112        assertThat(sObj2Arr2.get(0).toString(), equalTo("arr_value1"));
113        assertThat(sObj2Arr2.get(1).toString(), equalTo("arr_value2"));
114
115        sObj2Arr2 = (List) response.getHits().getAt(0).field("s_obj2_arr2").value();
116        assertThat(sObj2Arr2.size(), equalTo(2));
117        assertThat(sObj2Arr2.get(0).toString(), equalTo("arr_value1"));
118        assertThat(sObj2Arr2.get(1).toString(), equalTo("arr_value2"));
119    }

121    @Test
122    public void testFunctionScriptBoost() throws Exception {
123        wipeIndices("test");
124        createIndex("test");
125        index("test", "type1", "1", jsonBuilder().startObject().field("test", "value beck").field("num1", 1.0f).endObject());
126        index("test", "type1", "2", jsonBuilder().startObject().field("test", "value beck").field("num1", 2.0f).endObject());
127        refresh();
128
129        logger.info("--- QUERY_THEN_FETCH");

131        logger.info(" --> running doc['num1'].value");
