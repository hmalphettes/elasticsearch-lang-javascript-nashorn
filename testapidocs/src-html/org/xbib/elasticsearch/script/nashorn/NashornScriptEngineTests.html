039    @Test
040    public void testSimpleEquation() {
041        Map<String, Object> vars = new HashMap<String, Object>();
042        Object o = se.execute(se.compile("1 + 2"), vars);
043        assertThat(((Number) o).intValue(), equalTo(3));
044    }
045
046    @Test
047    public void testMapAccess() {
048        Map<String, Object> vars = new HashMap<String, Object>();
049
050        Map<String, Object> obj2 = MapBuilder.<String, Object>newMapBuilder().put("prop2", "value2").map();
051        Map<String, Object> obj1 = MapBuilder.<String, Object>newMapBuilder().put("prop1", "value1").put("obj2", obj2).put("l", Lists.newArrayList("2", "1")).map();
052        vars.put("obj1", obj1);
053        Object o = se.execute(se.compile("obj1"), vars);
054        assertThat(o, instanceOf(Map.class));
055        obj1 = (Map<String, Object>) o;
056        assertThat((String) obj1.get("prop1"), equalTo("value1"));
057        assertThat((String) ((Map<String, Object>) obj1.get("obj2")).get("prop2"), equalTo("value2"));
058
059        o = se.execute(se.compile("obj1.l[0]"), vars);
060        assertThat(((String) o), equalTo("2"));
061    }
062
063    @Test
064    public void testJavaScriptObjectToMap() {
065        Map<String, Object> vars = new HashMap<String, Object>();
066        Object o = se.execute(se.compile("var obj1 = {}; obj1.prop1 = 'value1'; obj1.obj2 = {}; obj1.obj2.prop2 = 'value2'; obj1"), vars);
067        Map obj1 = (Map) o;
068        assertThat((String) obj1.get("prop1"), equalTo("value1"));
069        assertThat((String) ((Map<String, Object>) obj1.get("obj2")).get("prop2"), equalTo("value2"));
070    }
071
072    @Test
073    public void testJavaScriptObjectMapInter() {
074        Map<String, Object> vars = new HashMap<String, Object>();
075        Map<String, Object> ctx = new HashMap<String, Object>();
076        Map<String, Object> obj1 = new HashMap<String, Object>();
077        obj1.put("prop1", "value1");
078        ctx.put("obj1", obj1);
079        vars.put("ctx", ctx);
080
081        se.execute(se.compile("ctx.obj2 = {}; ctx.obj2.prop2 = 'value2'; ctx.obj1.prop1 = 'uvalue1'"), vars);
082        ctx = (Map<String, Object>) se.unwrap(vars.get("ctx"));
083        assertThat(ctx.containsKey("obj1"), equalTo(true));
084        assertThat((String) ((Map<String, Object>) ctx.get("obj1")).get("prop1"), equalTo("uvalue1"));
085        assertThat(ctx.containsKey("obj2"), equalTo(true));
086        assertThat((String) ((Map<String, Object>) ctx.get("obj2")).get("prop2"), equalTo("value2"));
087    }
088
089    @Test
090    public void testJavaScriptInnerArrayCreation() {
091        Map<String, Object> ctx = new HashMap<String, Object>();
092        Map<String, Object> doc = new HashMap<String, Object>();
093        ctx.put("doc", doc);
094
095        Object complied = se.compile("ctx.doc.field1 = ['value1', 'value2']");
096        ExecutableScript script = se.executable(complied, new HashMap<String, Object>());
097        script.setNextVar("ctx", ctx);
098        script.run();
099
100        Map<String, Object> unwrap = (Map<String, Object>) script.unwrap(ctx);
101
102        assertThat(((Map) unwrap.get("doc")).get("field1"), instanceOf(List.class));
103    }
104
105    @Test
106    public void testAccessListInScript() {
107        Map<String, Object> vars = new HashMap<String, Object>();
108        Map<String, Object> obj2 = MapBuilder.<String, Object>newMapBuilder().put("prop2", "value2").map();
109        Map<String, Object> obj1 = MapBuilder.<String, Object>newMapBuilder().put("prop1", "value1").put("obj2", obj2).map();
110        vars.put("l", Lists.newArrayList("1", "2", "3", obj1));
111
112        Object o = se.execute(se.compile("l.length"), vars);
113        assertThat(((Number) o).intValue(), equalTo(4));
114
115        o = se.execute(se.compile("l[0]"), vars);
116        assertThat(((String) o), equalTo("1"));
117
118        o = se.execute(se.compile("l[3]"), vars);
119        obj1 = (Map<String, Object>) o;
120        assertThat((String) obj1.get("prop1"), equalTo("value1"));
121        assertThat((String) ((Map<String, Object>) obj1.get("obj2")).get("prop2"), equalTo("value2"));
122
123        o = se.execute(se.compile("l[3].prop1"), vars);
124        assertThat(((String) o), equalTo("value1"));
125    }

127    @Test
128    public void testChangingVarsCrossExecution1() {
129        Map<String, Object> vars = new HashMap<String, Object>();
130        Map<String, Object> ctx = new HashMap<String, Object>();
131        vars.put("ctx", ctx);
132        Object compiledScript = se.compile("ctx.value");
133
134        ExecutableScript script = se.executable(compiledScript, vars);
135        ctx.put("value", 1);
136        Object o = script.run();
137        assertThat(((Number) o).intValue(), equalTo(1));
138
139        ctx.put("value", 2);
140        o = script.run();
141        assertThat(((Number) o).intValue(), equalTo(2));
142    }

145    public void testChangingVarsCrossExecution2() {
146        Map<String, Object> vars = new HashMap<String, Object>();
147        Object compiledScript = se.compile("value");
148
149        ExecutableScript script = se.executable(compiledScript, vars);
150        script.setNextVar("value", 1);

154        script.setNextVar("value", 2);
